load_package f5;



system := {x**2*y*z**4 - t,
   x**5*y**7 - z**2*u,
   -x**3*z*v + y**2,
   -z**5*w + x*y**3}$



vars := {w, v, u, t, z, y, x}$


torder(reverse vars, revgradlex)$



gb := f5(system)$



end;
